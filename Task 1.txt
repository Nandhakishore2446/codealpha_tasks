import random

# List of 5 predefined words
word_list = ['apple', 'tiger', 'house', 'table', 'river']
secret_word = random.choice(word_list)

# Game state
guessed_letters = []
wrong_guesses = 0
max_wrong_guesses = 6

# Create display with underscores for unguessed letters
display_word = ['_' for _ in secret_word]

print("ðŸŽ® Welcome to Hangman!")
print("Guess the word, one letter at a time.")

while wrong_guesses < max_wrong_guesses and '_' in display_word:
    print("\nWord: ", ' '.join(display_word))
    guess = input("Enter a letter: ").lower()

    if not guess.isalpha() or len(guess) != 1:
        print("âš  Please enter a single alphabet letter.")
        continue

    if guess in guessed_letters:
        print("â— You already guessed that letter.")
        continue

    guessed_letters.append(guess)

    if guess in secret_word:
        print("âœ… Good guess!")
        for i in range(len(secret_word)):
            if secret_word[i] == guess:
                display_word[i] = guess
    else:
        wrong_guesses += 1
        print(f"âŒ Wrong guess! {max_wrong_guesses - wrong_guesses} tries left.")

# Game result
if '_' not in display_word:
    print("\nðŸŽ‰ Congratulations! You guessed the word:", secret_word)
else:
    print("\nðŸ’€ Game Over! The word was:", secret_word)
